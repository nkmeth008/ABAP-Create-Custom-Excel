FORM get_data.

  SELECT * FROM zfu_talep_table
      INTO TABLE gt_tablo.



  CREATE OBJECT go_alv
    EXPORTING
      i_parent = go_cont.

  CALL FUNCTION 'LVC_FIELDCATALOG_MERGE'
    EXPORTING
      i_structure_name = 'ZFU_TALEP_ST'

    CHANGING
      ct_fieldcat      = gt_fcat.

  wa_layout-col_opt = 'X'.
  wa_layout-edit = 'X'.


  CALL METHOD go_alv->set_table_for_first_display
    EXPORTING
      i_structure_name = 'ZFU_TALEP_ST'
      is_layout        = wa_layout
    CHANGING
      it_outtab        = gt_tablo
      it_fieldcatalog  = gt_fcat.

  CALL METHOD go_alv->register_edit_event
    EXPORTING
      i_event_id = cl_gui_alv_grid=>mc_evt_modified.

  CALL METHOD go_alv->refresh_table_display.

ENDFORM.
